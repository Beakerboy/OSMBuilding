/**

 * <osm ....>
 *   <node id="{nid}" lat="{lat}" lon="{lon}"/>
 *   <way id="{wid}"/>
 *     <nd ref="{nid}"/>
 *     <tag k="{key}" v="{value}"/>
 *   </way>
 * </osm>

$node_data = [];
foreach ($xml_response['osm'] as $data} {
    if ($way_data->type == 'node') {
      $node_id = $data->id;
      $node_data[$node_id] = [$data->lat, $data->lon];
    }
}

}
